print("\n--- 2. Punktschtzer & Konfidenzintervalle (Bootstrap) ---")

# 1. Punktschtzer berechnen
point_estimate = np.mean(data_values)

# 2. Bootstrap-Verfahren (gem VL6)
# Wir nutzen Bootstrap, da unsere Daten (aus NB02 bekannt) nicht perfekt normalverteilt sind.
n_bootstrap = 10000
bootstrap_means = []
rng = np.random.default_rng(42)

for _ in range(n_bootstrap):
    # Resampling aus den bereits bereinigten Daten
    sample = rng.choice(data_values, size=len(data_values), replace=True)
    bootstrap_means.append(np.mean(sample))

# 3. KI berechnen
alpha = 0.05
lower_bound = np.percentile(bootstrap_means, (alpha/2) * 100)
upper_bound = np.percentile(bootstrap_means, (1 - alpha/2) * 100)

# Ausgabe & Plot
print(f"Punktschtzer (Mittelwert): {point_estimate:.2f}")
print(f"95% Bootstrap-Konfidenzintervall: [{lower_bound:.2f}, {upper_bound:.2f}]")
print(f"\nInterpretation: Wir sind zu 95% sicher, dass der wahre Mittelwert der Population in diesem Intervall liegt.")
print(f"Da wir Bootstrapping genutzt haben, ist dieses Ergebnis robust gegenber der Schiefe unserer Daten.")

plt.figure(figsize=(10, 5))
sns.histplot(bootstrap_means, kde=True, color="skyblue", edgecolor="none")
plt.axvline(point_estimate, color='k', linestyle='--', label=f'Punktschtzer ({point_estimate:.2f})')
plt.axvline(lower_bound, color='red', linestyle=':', label='95% KI Grenzen')
plt.axvline(upper_bound, color='red', linestyle=':')
plt.title(f"Bootstrap-Verteilung des Mittelwerts fr '{target_var_label}'")
plt.xlabel("Geschtzter Mittelwert")
plt.legend()
plt.show()
